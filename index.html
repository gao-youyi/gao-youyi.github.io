/*----------------------------------------------------------------
广义表：
所有的有限个元素的序列，其中每一个元素作为原子项，可以是一个表，也可以是一个简单的结点，元素
表头就是当表部位空时，第一个元素，其余的所有的元素都被叫做表尾，例如：L(a,(b,c,d),(e,f,g))
那么她的表尾就是((b,c,d),(e,f,g)),作为整体是一个子表，作为L的表尾，表的长度是3，表头是(a)，
也是一个子表。
广义表的元素有相对次序
长度是最外层括号包含的元素的个数
深度定义为所包含的括号的重数
且规定：原子的深度为0，空表的深度为1；
可以共享
可以是一个递归的表



表结点{
    标志：tag =1,
    表头指针：hp(head point),
    表尾指针：tp(tail point),
}
原子结点{
    标志：tag =0,
    值域： atom,
}
*/

/*
联合：
存储：
1，所有的成员共享一个空间
2，同一时间只有一个成员是有效的
3，union的大小是其最大的成员
初始化：
1，对第一个成员进行初始化就好
*/
//联合常用的地方
#include <stdio.h>
typedef union
{
    int i;
    char ch[sizeof(int)]; //此时数组和i的大小一样，里面会存储i的值
} CHI;                    //这里i 和ch同时占据相同的地方，但是同一时刻只有一个数据是有效的
int main(int argc, char argv[])
{ //这个程序的作用就是显示1024这样的东西在内存中占据的空间或者存储方式
    CHI chi;
    int i;
    chi.i = 1024;
    for (i = 0; i < sizeof(int); i++)
    {
        printf("%02hhX", chi.ch[i]);
    }
    printf("\n");
    return 0;
}
//----------------------------------------------------------------
//广义表的存储表示
enum ElemTag
{
    ATOM, //ATOM==0,原子；LIST==1，子表；
    LIST
};
typedef struct GLNode
{
    ElemTag tag; //区分是原子还是子表，如果原子结点就是0，如果是子表就是1
    union        //这里是原子结点和表结点的联合部分，共同使用
    {
        char atom; //atom的数据类型可以用户自定义
        struct     //这里是子表的数据类型，
        {
            GLNode *hp, *tp; //两个指针
        } ptr;               //起个别名，使得ptr.hp指向表头，ptr.tp指向表尾，这里有点递归定义的感觉
    };
} * GList, GLNode; //广义表的数据类型，可以在脑海里想象一下无数个表和原子结点的嵌套

//----------------------------------------------------------------

//拓展线性链表的存储结构
//此时会i形成一条更加简洁的单链，即使是原子结点，仍然可以通过tp指针指向下一个结点(可以是原子结点，也可以是子表结点)
enum ElemTag
{
    ATOM,
    LIST
};
typedef struct GLNode1
{
    ElemTag tag; //
    union
    {
        char atom;
        GLNode *hp;
    };
    GLNode1 *tp;
} * GList1, GLNode1;
